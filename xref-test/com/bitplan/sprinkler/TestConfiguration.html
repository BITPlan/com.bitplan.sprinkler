21  package com.bitplan.sprinkler;
22  
23  import static org.junit.Assert.assertEquals;
24  import static org.junit.Assert.assertNotNull;
25  
26  import org.junit.Test;
27  import org.openweathermap.weather.Coord;
28  import org.openweathermap.weather.Location;
29  
30  /**
31   * check the configuration
32   * @author wf
33   *
34   */
35  public class TestConfiguration {
36  
37    @Test
38    public void testConfiguration() throws Exception {
39      Configuration conf=new Configuration();
40      Location loc=Location.byName("Knickelsdorf/DE");
41      conf.setLocation(loc);
42      conf.setAreaSizeSquareMeter(100);
43      conf.setEarliestSprinkleHour("07:30");
44      conf.setLatestSprinkleHour("20:30");
45      conf.setlEvaporationPerDay(200);
46      conf.setSprinklesPerDay(2);
47      conf.setMmPerHour(22.8); // 2280 l per hour
48      Configuration conf2=new Configuration();
49      Location loc2=new Location();
50      Coord coord=new Coord(50.94,6.958);
51      loc2.setCoord(coord);
52      loc2.setName("Krefeld");
53      loc2.setCountry("DE");
54      loc2.setId(Location.byName("Krefeld/DE").getId());
55      conf2.setLocation(loc2);
56      Location loc3=Location.byId(loc2.getId());
57      // check results
58      assertEquals(loc3.getName(),loc2.getName());
59      assertEquals("DE",loc3.getCountry()); 
60      assertEquals(100,conf.getAreaSizeSquareMeter(),0.1);
61      assertEquals("07:30",conf.getEarliestSprinkleHour());
62      assertEquals("20:30",conf.getLatestSprinkleHour());
63      assertEquals(200,conf.getlEvaporationPerDay(),0.01);
64      assertEquals(2,conf.getSprinklesPerDay());
65      assertEquals(22.8,conf.getMmPerHour(),0.01);
66      assertNotNull(conf.getLocation());
67    }
68  
69  }
