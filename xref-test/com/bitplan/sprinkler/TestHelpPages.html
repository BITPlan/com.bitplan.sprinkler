50    /**
51     * start the Sprinkler software
52     * 
53     * @return nothing
54     */
55    public Void startSprinkler() {
56      Sprinkler.testMode = true;
57      sprinkler = new Sprinkler();
58      String[] args = {};
59      sprinkler.maininstance(args);
60      return null;
61    }

63    /**
64     * create a snapShot
65     * 
66     * @param stage
67     * @param title
68     */
69    public void snapShot(Stage stage, String title) {
70      // PauseTransition pause = new PauseTransition(Duration.millis(500));
71      // pause.setOnFinished(event -> {
72      try {
73        File snapShotDir = new File("/tmp/sprinkler");
74        snapShotDir.mkdirs();
75        title = title.replace("/", "");
76        File snapShot = new File(snapShotDir, title + ".png");
77        if (debug)
78          System.out.println(snapShot.getAbsolutePath());
79        WaitableApp.saveAsPng(stage, snapShot);
80      } catch (Exception e) {
81        e.printStackTrace();;
82      }
83  
84      // });
85      // pause.play();
86    }
87  
88    boolean done = false;
89  
90    public void loopTabs(GenericApp display) {
91      XYTabPane xyTabPane = display.getXyTabPane();
92      ObservableList<Tab> vtabs = xyTabPane.getvTabPane().getTabs();
93      for (Tab vtab : vtabs) {
94        xyTabPane.getvTabPane().getSelectionModel().select(vtab);
95        TabPane hTabPane = xyTabPane.getSelectedTabPane();
96        ObservableList<Tab> htabs = hTabPane.getTabs();
97  
98        if (vtab.getTooltip() != null) {
99          String vTitle = vtab.getTooltip().getText();
100         for (Tab htab : htabs) {
101           hTabPane.getSelectionModel().select(htab);
102           String title = htab.getTooltip().getText();
103           System.out.println(vTitle + "_" + title);
104           snapShot(display.getStage(), vTitle + "_" + title);
105         }
106       }
107     }
108     done = true;
109 
110   }

112   @Test
113   public void testTabs() throws Exception {
114     WaitableApp.toolkitInit();
115     WaitableApp.testMode=true;
116     TaskLaunch.start(() -> startSprinkler());
117     while (sprinkler == null || sprinkler.gApp == null)
118       Thread.sleep(10);
119     GenericApp gApp = sprinkler.gApp;
120     gApp.waitOpen();
121     Thread.sleep(250);
122     Platform.runLater(() -> loopTabs(gApp));
123     while (!done)
124       Thread.sleep(50);
125   }
