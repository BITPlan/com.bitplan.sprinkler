21  package com.bitplan.sprinkler.javafx.presenter;
22  
23  import java.io.IOException;
24  
25  import com.bitplan.appconfig.Preferences;
26  import com.bitplan.appconfig.Preferences.LangChoice;
27  import com.bitplan.error.ExceptionHandler;
28  import com.bitplan.gui.App;
29  import com.bitplan.i18n.I18n;
30  import com.bitplan.i18n.Translator;
31  import com.bitplan.javafx.BaseModifier;
32  import com.bitplan.javafx.GenericDialog;
33  import com.bitplan.javafx.GenericPanel;
34  import com.bitplan.sprinkler.SprinklerI18n;
35  
36  import javafx.stage.Stage;
37  
38  
39  /**
40   * present the preferences
41   * @author wf
42   *
43   */
44  @SuppressWarnings("restriction")
45  public class PreferencesModifier extends BaseModifier<Preferences>{
46    
47    /**
48     * construct me
49     * @param preferencesPanel 
50     * @param app 
51     * @param stage 
52     * @param preferences
53     */
54    public PreferencesModifier(Stage stage, App app, ExceptionHandler exceptionHandler,GenericPanel preferencesPanel, Preferences preferences) {
55      super.init(stage, app, exceptionHandler);
56      this.setModel(preferences);
57      this.setView(preferencesPanel);
58      updateView();
59    }
60  
61    @Override
62    public void updateView() {
63      this.getView().setValues(this.getModel().asMap());
64    }
65  
66    @Override
67    public Preferences updateModel() {
68      Preferences preferences=this.getModel();
69      LangChoice lang = preferences.getLanguage();
70      preferences.fromMap(this.getView().getValueMap());
71      try {
72        preferences.save();
73      } catch (IOException e) {
74        this.getExceptionHandler().handleException(e);
75      }
76      if (!lang.equals(preferences.getLanguage())) {
77        Translator.initialize(Translator.APPLICATION_PREFIX, preferences.getLanguage().name());
78        GenericDialog.showAlert(getStage(), I18n.get(SprinklerI18n.LANGUAGE_CHANGED_TITLE),
79            I18n.get(SprinklerI18n.LANGUAGE_CHANGED), I18n.get(SprinklerI18n.NEWLANGUAGE_RESTART));
80      }
81      return preferences;
82    }
83  
84  }
